Full Rollback Command Playbook
# ── STEP 1: Spot the broken deployment ──────────────────
$ kubectl rollout history deployment/myapp
REVISION CHANGE-CAUSE
1 Initial release v1.0.0
2 Broken DB migration v2.0.0

# ── STEP 2: Inspect the bad revision ───────────────────
$ kubectl rollout history deployment/myapp --revision=2
Image: myapp:2.0.0
Labels: app=myapp, version=2.0.0

# ── STEP 3: Rollback to v1 (revision 1) ────────────────
$ kubectl rollout undo deployment/myapp --to-revision=1
deployment.apps/myapp rolled back

# ── STEP 4: Monitor recovery ────────────────────────────
$ kubectl rollout status deployment/myapp
deployment "myapp" successfully rolled out

# ── STEP 5: Verify history after rollback ───────────────
$ kubectl rollout history deployment/myapp
REVISION CHANGE-CAUSE
2 Broken DB migration v2.0.0
3 Initial release v1.0.0 ← re-applied via rollback

# ── STEP 6: Deploy fixed v3 after hotfix ────────────────
$ kubectl set image deployment/myapp app=myapp:3.0.0
deployment.apps/myapp image updated
$ kubectl annotate deployment/myapp kubernetes.io/change-cause="Hotfix v3.0.0"
